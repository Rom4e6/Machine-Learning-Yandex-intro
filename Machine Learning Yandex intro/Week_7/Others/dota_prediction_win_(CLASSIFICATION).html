Также есть и другие подходы — например, замена пропуска на среднее значение признака. def find_best_clf(model, X_train, y_train, params, scoring):
    # model: классификатор исполюзующийся для поиска лучших гипер параметров
    # X_train, y_train: np.array, массивы датасета
    # params: dict or list значения параметров классификатора
    # scoring: метрика качества используемого классификатора
    # Функция возвращает классификатор с лучшими params, и лучшей метрикой качества scoring

    # Лучший классификатор GradientBoosting = n_estimators: 300, но использовать будем 30, т.к. получаемое качество
    # не сильно отличается от 300 деревьев, а это влияет на скорость обучения.

    grid_search = GridSearchCV(model,
                               params,
                               scoring=scoring)
    grid_search.fit(X_train, y_train)
    return grid_search.best_estimator_

def fit_model(model, X_train, y_train, X_test):
    # Функция принимает на вход:
    # X_train, y_train: np.array, обучающей выборки
    # X_test: np.array, тестовой выборки
    # Функция возращает прогноз для тестовой выборки 1 индекса метода predict_proba

    model.fit(X_train, y_train)
    predict_proba = model.predict_proba(X_test)[:, 1]
    return predict_proba

def graph(prediction_model):
    # Построение графика полученного прогноза
    # prediction_model: np.array

    preds = pd.Series(prediction_model)
    preds.describe()
    preds.plot.hist(bins=30)
    plt.show()

def cross_scores(list_estimators, X_train, y_train, scoring=None, cv=None):
    # list_estimators: list, кол-во деревьев принимаемых
    # X_train, y_train: np.array, обучающей выборки
    # scoring: str, метрика оцениваемого качесва
    # cv: KFold генератор
    # Функция проводит кросс валидацию классификатора и строит на выходе график полученных значений качества

    scores = {}
    for n_estimator in list_estimators:
        model = GradientBoostingClassifier(n_estimators=n_estimator)
        start_time = datetime.datetime.now()
        score = cross_val_score(model, X_train, y_train, scoring=scoring, cv=cv).mean()
        print('Estimators: {}'.format(n_estimator))
        print('Score: {}'.format(score))
        print('Script work time: {}'.format(datetime.datetime.now() - start_time))
        scores[n_estimator] = score

    return pd.Series(scores)

def cross_scores_lrg(list_c, X_train, y_train, scoring=None, cv=None):
    # list_c: list, кол-во деревьев принимаемых
    # X_train, y_train: np.array, обучающей выборки
    # scoring: str, метрика оцениваемого качесва
    # cv: KFold генератор
    # Функция проводит кросс валидацию классификатора и строит на выходе график полученных значений качества

    scores = {}
    for c in list_c:
        model = LogisticRegression(C=c)
        start_time = datetime.datetime.now()
        score = cross_val_score(model, X_train, y_train, scoring=scoring, cv=cv).mean()
        print('C: {}'.format(c))
        print('Score: {}'.format(score))
        print('Script work time: {}'.format(datetime.datetime.now() - start_time))
        scores[c] = score

    return pd.Series(scores)

def get_pick(data):
    # data: pd.DataFrame, обучающей и тестовой выборки
    # Функция возвращает 113 дополнительных признаков, для каждого героя участвовавшего в матче или нет
    
    X_pick = np.zeros((data.shape[0], 113))

    for i, match_id in enumerate(data.index):
        for p in range(1, 6):
            X_pick[i, data.loc[match_id, f"r{p}_hero"] - 1] = 1
            X_pick[i, data.loc[match_id, f"d{p}_hero"] - 1] = -1

    return pd.DataFrame(X_pick, index=data.index, columns=[f"hero_{i}" for i in range(113)])
